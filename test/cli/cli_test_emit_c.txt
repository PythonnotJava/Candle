/* class Greeter */
typedef struct Greeter Greeter;
struct Greeter {
    candle_string name;
    candle_int count;
};

static Greeter *Greeter__new(candle_string name, candle_int count) {
    Greeter *self = GC_MALLOC(sizeof(Greeter));
    self->name = name;
    self->count = count;
    return self;
}

    print(candle_str_concat(candle_str_concat(candle_str("[Greeter] Hello, "), msg), candle_str("!")));
    self->count = (self->count + ((candle_int)1));
}

    return (a + b);
}
